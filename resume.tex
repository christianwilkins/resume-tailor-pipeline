% Custom commands
\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}}
  }
}

% Simplified resume subheading for better ATS parsing
% #1: Title (e.g., Company Name or Degree)
% #2: Date Range
% #3: Subtitle (e.g., Role or Major)
% #4: Location
\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
  \textbf{#1} \hfill #2 \\
  \textit{\small#3} \hfill \textit{\small #4} \\
  \vspace{-7pt} % Adjust vertical spacing as needed
}

% Project heading (simplified for ATS)
% #1: Project Title and Technologies
% #2: Date Range
\newcommand{\resumeProjectHeading}[2]{
  \item
  \small#1 \hfill #2 \\
  \vspace{-7pt} % Adjust vertical spacing as needed
}

% Added this to remove 2nd line.
\newcommand{\resumeSubheadingTwo}[2]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
    \end{tabular*}\vspace{-5pt}
}

\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}
